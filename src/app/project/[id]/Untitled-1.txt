            grid.rotation.x = -Math.PI / 2;
            grid.position.y = -2;
            scene.add(grid);

            // Add some floating cubes
            const cubes = [];
            const cubeGeo = new THREE.BoxGeometry(0.5, 0.5, 0.5);
            const cubeMat = new THREE.MeshBasicMaterial({ color: 0xbc0100, wireframe: true, transparent: true, opacity: 0.3 });
            
            for(let i=0; i<10; i++) {
                const cube = new THREE.Mesh(cubeGeo, cubeMat);
                cube.position.set((Math.random() - 0.5) * 10, Math.random() * 5, (Math.random() - 0.5) * 10);
                scene.add(cube);
                cubes.push(cube);
            }

            camera.position.z = 5;
            camera.position.y = 1;

            const animate = () => {
                requestAnimationFrame(animate);
                grid.position.z += 0.01;
                if(grid.position.z > 1) grid.position.z = 0;
                
                cubes.forEach((cube, i) => {
                    cube.rotation.x += 0.01;
                    cube.rotation.y += 0.01;
                    cube.position.y += Math.sin(Date.now() * 0.001 + i) * 0.005;
                });
                
                renderer.render(scene, camera);
            };

            animate();

            window.addEventListener('resize', () => {
                camera.aspect = container.offsetWidth / container.offsetHeight;
                camera.updateProjectionMatrix();
                renderer.setSize(container.offsetWidth, container.offsetHeight);
            });
        };

        window.onload = initThree;

        // Intersection Observer for assemble animations
        const observerOptions = {
            threshold: 0.1,
            rootMargin: "0px 0px -100px 0px"
        };

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('visible');
                }
            });
        }, observerOptions);

        document.querySelectorAll('.assemble-block, .assemble-line-x, .assemble-line-y').forEach((el) => {
            observer.observe(el);
        });
        
        // Side Nav Scroll Spy
        const sections = document.querySelectorAll('section[id]');
        const navDots = document.querySelectorAll('.nav-dot');
        const progressBar = document.getElementById('nav-progress');
        
        window.addEventListener('scroll', () => {
            let current = '';
            const scrollY = window.pageYOffset;
            
            // Calculate progress bar height based on scroll
            const totalHeight = document.documentElement.scrollHeight - window.innerHeight;
            const progress = (scrollY / totalHeight) * 100;
            if(progressBar) progressBar.style.height = `${progress}%`;
            
            sections.forEach(section => {
                const sectionTop = section.offsetTop;
                const sectionHeight = section.clientHeight;
                if (scrollY >= (sectionTop - sectionHeight / 3)) {
                    current = section.getAttribute('id');
                }
            });
            
            navDots.forEach(dot => {
                dot.classList.remove('active');
                if (dot.getAttribute('href').includes(current)) {
                    dot.classList.add('active');
                }
            });
        });
    </script>
